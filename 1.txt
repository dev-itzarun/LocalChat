The current updater has a fundamental problem — it **manually copies files** instead of using VS Code's own installer. This is the wrong approach entirely. Here's what to change and why:

---

## The Core Problem

```js
// ❌ Current approach — copies raw files, bypasses the editor completely
copyDir(srcExtension, extensionPath);
```

This breaks because:
- VS Code/Antigravity/Cursor track extension state internally — raw file copies don't register
- The extension folder may be read-only or sandboxed depending on OS/editor
- Partially copied files leave the extension broken with no recovery
- It doesn't work at all on some VS Code forks

---

## The Right Approach

Use the editor's **own built-in command** to install the downloaded VSIX — the same thing that happens when you run `code --install-extension file.vsix`. The API for this is:

```js
vscode.commands.executeCommand('workbench.extensions.installExtension', vscode.Uri.file(vsixPath))
```

This works on VS Code, Antigravity, Cursor, and all forks because it's a core workbench command.

---

## Full Rewrite of `updater.js`

**Replace the entire file with this:**

```js
// src/updater.js — Self-update from GitHub Releases
// Downloads VSIX to temp, installs via VS Code's own installer command.
// Works on VS Code, Antigravity, Cursor, and any VS Code fork.

const https  = require('https');
const http   = require('http');
const fs     = require('fs');
const path   = require('path');
const os     = require('os');
const vscode = require('vscode');

const REPO           = 'triode-devs/local-chat';
const API_URL        = `https://api.github.com/repos/${REPO}/releases/latest`;
const CHECK_INTERVAL = 4 * 60 * 60 * 1000; // 4 hours
const CURRENT_VER    = require('../package.json').version;

// ── Semver compare ────────────────────────────────────
function isNewer(latest, current) {
  const parse = v => v.replace(/^v/, '').split('.').map(Number);
  const [la, lb, lc] = parse(latest);
  const [ca, cb, cc] = parse(current);
  return la > ca || (la === ca && lb > cb) || (la === ca && lb === cb && lc > cc);
}

// ── HTTPS GET with redirect limit ─────────────────────
function get(url, opts = {}, _hops = 0) {
  return new Promise((resolve, reject) => {
    if (_hops > 5) return reject(new Error('Too many redirects'));
    const mod = url.startsWith('https') ? https : http;
    const req = mod.get(
      url,
      { ...opts, headers: { 'User-Agent': 'local-chat-updater', ...(opts.headers || {}) } },
      res => {
        if (res.statusCode >= 300 && res.statusCode < 400 && res.headers.location)
          return get(res.headers.location, opts, _hops + 1).then(resolve).catch(reject);
        let data = '';
        res.on('data', c => data += c);
        res.on('end', () => resolve({ status: res.statusCode, body: data }));
      }
    );
    req.on('error', reject);
    req.setTimeout(10000, () => { req.destroy(); reject(new Error('Request timed out')); });
  });
}

// ── Download file to disk with redirect limit ─────────
function download(url, dest, _hops = 0) {
  return new Promise((resolve, reject) => {
    if (_hops > 5) return reject(new Error('Too many redirects'));
    const mod = url.startsWith('https') ? https : http;
    const req = mod.get(
      url,
      { headers: { 'User-Agent': 'local-chat-updater' } },
      res => {
        if (res.statusCode >= 300 && res.statusCode < 400 && res.headers.location)
          return download(res.headers.location, dest, _hops + 1).then(resolve).catch(reject);
        if (res.statusCode !== 200)
          return reject(new Error(`Download failed: HTTP ${res.statusCode}`));

        const tmp  = dest + '.part';          // write to .part first
        const file = fs.createWriteStream(tmp);
        res.pipe(file);
        file.on('finish', () => {
          file.close(() => {
            fs.renameSync(tmp, dest);         // atomic rename on finish
            resolve();
          });
        });
        file.on('error', err => {
          fs.unlink(tmp, () => {});
          reject(err);
        });
      }
    );
    req.on('error', reject);
    req.setTimeout(60000, () => { req.destroy(); reject(new Error('Download timed out')); });
  });
}

// ── Main update check ─────────────────────────────────
async function checkForUpdate(silent = false) {
  try {
    const { status, body } = await get(API_URL);
    if (status !== 200) return;

    const release = JSON.parse(body);
    const latest  = release.tag_name?.replace(/^v/, '');

    if (!latest || !isNewer(latest, CURRENT_VER)) {
      if (!silent)
        vscode.window.showInformationMessage(`Local Chat v${CURRENT_VER} is up to date.`);
      return;
    }

    // Find .vsix asset
    const asset = release.assets?.find(a => a.name.endsWith('.vsix'));
    if (!asset) {
      if (!silent)
        vscode.window.showWarningMessage(`Local Chat v${latest} found but no VSIX asset attached to the release.`);
      return;
    }

    const action = await vscode.window.showInformationMessage(
      `🆕 Local Chat v${latest} is available (you have v${CURRENT_VER})`,
      'Update Now',
      'Later'
    );
    if (action !== 'Update Now') return;

    await installUpdate(asset.browser_download_url, latest);

  } catch (err) {
    if (!silent)
      vscode.window.showErrorMessage(`Local Chat update check failed: ${err.message}`);
  }
}

// ── Install via VS Code's own VSIX installer ──────────
async function installUpdate(downloadUrl, newVersion) {
  // Use a stable, predictable temp path (not random) so stale .part files
  // from a previous crashed attempt are always cleaned up on next run.
  const tmpDir  = path.join(os.tmpdir(), 'local-chat-update');
  const vsixPath = path.join(tmpDir, `local-chat-${newVersion}.vsix`);

  // Clean up any previous attempt
  try { fs.rmSync(tmpDir, { recursive: true, force: true }); } catch (_) {}
  fs.mkdirSync(tmpDir, { recursive: true });

  try {
    await vscode.window.withProgress(
      {
        location: vscode.ProgressLocation.Notification,
        title: 'Local Chat: Updating…',
        cancellable: false,
      },
      async progress => {
        // Step 1 — Download
        progress.report({ message: 'Downloading…', increment: 30 });
        await download(downloadUrl, vsixPath);

        // Step 2 — Hand off to the editor's own installer
        // This is equivalent to running: code --install-extension file.vsix
        // Works on VS Code, Cursor, Antigravity, and all forks.
        progress.report({ message: 'Installing…', increment: 60 });
        await vscode.commands.executeCommand(
          'workbench.extensions.installExtension',
          vscode.Uri.file(vsixPath)
        );

        progress.report({ message: 'Done!', increment: 10 });
      }
    );

    // Prompt reload — the editor needs a restart to activate the new version
    const reload = await vscode.window.showInformationMessage(
      `✅ Local Chat updated to v${newVersion}! Reload to apply.`,
      'Reload Now',
      'Later'
    );
    if (reload === 'Reload Now')
      vscode.commands.executeCommand('workbench.action.reloadWindow');

  } catch (err) {
    vscode.window.showErrorMessage(`Local Chat update failed: ${err.message}`);
  } finally {
    // Always clean up the downloaded VSIX
    try { fs.rmSync(tmpDir, { recursive: true, force: true }); } catch (_) {}
  }
}

// ── Auto-update scheduler ─────────────────────────────
function startAutoUpdater() {
  // Delay first check by 60s so it doesn't slow down extension activation.
  // (The old code used 5s which fires on every restart.)
  setTimeout(() => checkForUpdate(true), 60 * 1000);
  setInterval(() => checkForUpdate(true), CHECK_INTERVAL);
}

module.exports = { startAutoUpdater, checkForUpdate };
```

---

## What Changed and Why

**Removed `extensionPath` parameter from everything**
The old code passed `extensionPath` around to do the manual file copy. None of that is needed anymore — the editor handles it. Update your call sites:

```js
// extension.js — BEFORE:
const { startAutoUpdater, checkForUpdate } = require('./src/updater');
startAutoUpdater(context.extensionPath);

// AFTER — no path needed:
startAutoUpdater();

// And for manual check command:
// BEFORE: checkForUpdate(context.extensionPath, false)
// AFTER:
checkForUpdate(false)
```

**Removed `extractVsix` and `copyDir` entirely**
Those 40+ lines of platform-specific extraction code are gone. The editor's own installer handles extraction correctly on every platform.

**Download uses `.part` file + atomic rename**
```js
const tmp = dest + '.part';
// write to .part...
fs.renameSync(tmp, dest);  // only renamed when complete
```
If the download is interrupted, there's no corrupt partial file left behind at the final path.

**Temp directory is now deterministic, not random**
```js
// BEFORE — new random dir every time, old ones pile up in /tmp:
fs.mkdtempSync(path.join(os.tmpdir(), 'local-chat-update-'))

// AFTER — fixed path, always cleaned before use:
path.join(os.tmpdir(), 'local-chat-update')
```

**Startup delay changed from 5s → 60s**
5 seconds means the update check fires on every single editor restart. 60 seconds gives the editor time to fully activate without the update HTTP request competing with startup.
